加入房间后，请增加 NERoom 房间的监听事件，便于及时获取相应的回调信息。

## 注意事项

需要在加入房间后进行事件监听。

## 配置步骤

1. 调用 `getRoomContext` 接口获取房间上下文。
    
    调用此方法时，您需要将 roomUid 设置为您所加入房间的 ID。

2. 调用 [`addEventCallback`](https://pub.dev/documentation/netease_roomkit/latest/netease_roomkit/NEPreviewRoomContext/addEventCallback.html) 接口监听房间里的事件。

## 示例代码

```dart
NERoomService roomService = NERoomKit.instance.roomService;
NERoomContext? roomContext = roomService.getRoomContext(roomUuid);
roomContext?.addEventCallback(NERoomEventCallback(
    roomEnd: (NERoomEndReason reason) {
    debugPrint('roomEnd');
    }, 
    roomPropertiesChanged: (Map<String, String> properties) {
    debugPrint('roomPropertiesChanged');
    },
    roomPropertiesDeleted: (Map<String, String> properties) {
    debugPrint('roomPropertiesDeleted');
  },
   memberPropertiesChanged:(NERoomMember member, Map<String, String> properties) {
    debugPrint('memberPropertiesChanged');
  },
   memberPropertiesDeleted:(NERoomMember member, Map<String, String> properties) {
    debugPrint('memberPropertiesDeleted');
  },
   memberRoleChanged:(NERoomMember member, NERoomRole before, NERoomRole after) {
    debugPrint('memberRoleChanged');
  },
   memberNameChanged: (NERoomMember member, String name) {
    debugPrint('memberNameChanged');
  },
   memberJoinRoom: (List<NERoomMember> members) {
    debugPrint('memberJoinRoom');
  }, 
  memberLeaveRoom: (List<NERoomMember> members) {
    debugPrint('memberLeaveRoom');
  },
   memberJoinRtcChannel: (List<NERoomMember> members) {
    debugPrint('memberJoinRtcChannel');
  },
   memberLeaveRtcChannel: (List<NERoomMember> members) {
    debugPrint('memberLeaveRtcChannel');
  }, 
  rtcChannelError: (int code) {
    debugPrint('rtcChannelError');
  },
   memberJoinChatroom: (List<NERoomMember> members) {
    debugPrint('memberJoinChatroom');
  }, 
  memberLeaveChatroom: (List<NERoomMember> members) {
    debugPrint('memberLeaveChatroom');
  }, 
  memberAudioMuteChanged:
          (NERoomMember member, bool mute, NERoomMember? operator) {
    debugPrint('memberAudioMuteChanged');
  },
   memberVideoMuteChanged:
          (NERoomMember member, bool mute, NERoomMember? operator) {
    debugPrint('memberVideoMuteChanged');
  },
   memberScreenShareStateChanged:
          (NERoomMember member, bool isSharing, NERoomMember? operator) {
    debugPrint('memberScreenShareStateChanged');
  },
   chatroomMessagesReceived: (List<NERoomMessage> messages) {
    debugPrint('chatroomMessagesReceived');
  },
   memberWhiteboardShareStateChanged:
          (NERoomMember member, bool isSharing, NERoomMember? operator) {
    debugPrint('memberWhiteboardShareStateChanged');
  },
   whiteboardError: (int code, String? msg) {
    debugPrint('whiteboardError');
  },
   roomLockStateChanged: (bool isLocked) {
    debugPrint('roomLockStateChanged');
  }, 
  liveStateChanged: (NELiveState state) {
    debugPrint('liveStateChanged');
  },
   rtcAudioVolumeIndication:
          (List<NEMemberVolumeInfo> volumes, int totalVolume) {
    debugPrint('rtcAudioVolumeIndication');
  },
   rtcAudioOutputDeviceChanged: (NEAudioOutputDevice device) {
    debugPrint('rtcAudioOutputDeviceChanged');
  }
  ));
```
